    public static <AnyType extends Comparable<? super AnyType>> int binary(AnyType[] a,int low,int high,AnyType x){
        
        while(low <= high){
            int mid = (low + high)/2;
            if(a[mid].compareTo(x) > 0)
                high = mid -1;
            else if(a[mid].compareTo(x) < 0)
                low = mid + 1;
            else

        }
        return -1;
    }
